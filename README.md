# flumelog-array

> like flumelog-memory without the filesystem

## Install

```
npm install flumelog-array
```

## Usage

```js
const Log = require('flumelog-array')
const Flume = require('flumedb')

const db = Flume(Log())

db.append({foo: 1}, function (err, seq) {
  if (err) throw err
  db.get(seq, (err, val) => {
    if (err) throw err
    console.log(val) // => { foo: 1 }
  })
})
```
